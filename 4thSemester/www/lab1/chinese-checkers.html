Chinese Checkers

Classic variant for 6 players

Implementation of Chinese Checkers written in Go.

Game has versions for 2, 3, 4 and 6 players and two variants: Classic with players stating in their corners and Chaos where all players start in the center. Implementation also has the option to add bots making random moves.

Backend is made in Go, while the front end mostly consists of TypeScript. The game is fully adapted to online play upon hosting the backend.

Here is a recursive function checking all possible jumps available to a bot:

func (b *Bot) jumpCheck(point1 Point, x, y int) {
pawns := b.board.GetPawns()

point2 := Point{x: x, y: y}
if slices.Contains(b.moves, [2]Point{point1, point2}) {
return
}

b.moves = append(b.moves, [2]Point{point1, point2})

if pawns.Check(x-2, y) != 0 && pawns.Check(x-4, y) == 0 && b.board.Check(x-4, y) >= 0 {
b.jumpCheck(point1, x-4, y)
}

if pawns.Check(x+2, y) != 0 && pawns.Check(x+4, y) == 0 && b.board.Check(x+4, y) >= 0 {
b.jumpCheck(point1, x+4, y)
}

if pawns.Check(x-1, y-1) != 0 && pawns.Check(x-2, y-2) == 0 && b.board.Check(x-2, y-2) >= 0 {
b.jumpCheck(point1, x-2, y-2)
}

if pawns.Check(x-1, y+1) != 0 && pawns.Check(x-2, y+2) == 0 && b.board.Check(x-2, y+2) >= 0 {
b.jumpCheck(point1, x-2, y+2)
}

if pawns.Check(x+1, y-1) != 0 && pawns.Check(x+2, y-2) == 0 && b.board.Check(x+2, y-2) >= 0 {
b.jumpCheck(point1, x+2, y-2)
}

if pawns.Check(x+1, y+1) != 0 && pawns.Check(x+2, y+2) == 0 && b.board.Check(x+2, y+2) >= 0 {
b.jumpCheck(point1, x+2, y+2)
}
}
